Problem 1

bool isSolvable(Cube & cube, CubeSet & seen) {
	if(isSolution(cube)) return true;
	if(seen.contains(cube)) continue;
	seen.add(cube);

	for(int i = 0; i < 8; i++) {
		// rotate clockwise
		rotate(cube, i, true);
		if(isSolvable(cube, seen)) return true;
		rotate(cube, i, false);

		// rotate counter clockwise
		rotate(cube, i, false);
		if(isSolvable(cube, seen)) return true;
		rotate(cube, i, true);

	}
	return false;
}

Base Case [4 points]

  • [4 points] Checks for (a) solution (b) seen set (c) updates seen set.
  • [3 points] Misses one of the three or confuses true with false.
  • [1 point] Gets only one of the three right
  • [0 point] No base case

Recursive Case [7 points]

  • [3 points] Checks for (a) solution (b) seen set (c) updates seen set.
  • [2 points] Misses one of the three or confuses true with false.
  • [1 point] Major errors
  • [0 point] No base case
